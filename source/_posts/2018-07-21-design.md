### **单列模式**
   单列模式 适合登录框这种场景或者需要全局通信的场景，不同页面调用其实都的是一个对象
   这样不会生成多个实例造成混乱
```
{
    //es6 单列模式
    class Login{
        constructor(state) {
            
        }
        login() {
            console.log('....login')
        }
    }
     //定义一个静态方法，这个方法是唯一的 
    Login.getInstance = (function(){
        let instance;
         return function() {
             if(!instance) {
                instance = new Login()
             }
             return instance
         }
    })()

    //测试
    let obj1 = new Login.getInstance();
    obj1.login();

    let obj2 = new Login.getInstance();
    obj2.login();

    
    console.log(obj1 === obj2)  // ...login  ...login   true

    let obj3 = new Login()
    console.log(obj1 === obj3)  //false 没有走静态方法，不可控


}




{
   //ES5 单列
   //对象字面量
   var Login = {
       status : false,
       login:function() {
          if(this.status == true) {
            console.log('已经登陆了')
            return 
          }
          this.status = true;
          console.log('....login')
       },
       logout:function() {
          if(this.status == false) {
            console.log('已经登出了')
            return 
          }
          this.status = false;
          console.log('....logout')
       }
   }
  
   var o1 = Login;
       o1.login();  //login....

   var o2 = Login
       o2.login()  //已经登录了
   console.log(o1 === o2)  //true
}



```



### **工厂模式**
下面这个模拟JQUERY的其实就是个工厂模式 ,JQuery这个大类提供API
我们用的时候直接使用 window上挂载的 $ 这个就相当于JQuery的工厂
不必关心工厂是怎么实现的
